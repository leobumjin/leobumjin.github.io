from __future__ import annotations
from typing import Optional

def format_chat(
    tokenizer,
    prompt: str,
    response: Optional[str] = None,
    add_generation_prompt: bool = False,
    system_instruction: Optional[str] = None,
) -> str:
    """
    Format a chat conversation using the tokenizer's chat template.

    Args:
        tokenizer:
            A tokenizer object that provides `apply_chat_template`.
        prompt:
            The user message.
        response:
            Optional assistant response to include in the conversation.
        add_generation_prompt:
            Whether to append the assistant generation prompt.
            Usually True when preparing model input for inference.
        system_instruction:
            Optional system message that defines global behavior.

    Returns:
        A formatted chat string.
    """

    if not isinstance(prompt, str) or not prompt.strip():
        raise ValueError("`prompt` must be a non-empty string.")

    if response is not None and not isinstance(response, str):
        raise TypeError("`response` must be a string or None.")

    if system_instruction is not None and not isinstance(system_instruction, str):
        raise TypeError("`system_instruction` must be a string or None.")

    messages = []

    if system_instruction is not None and system_instruction.strip():
        messages.append({"role": "system", "content": system_instruction.strip()})

    messages.append({"role": "user", "content": prompt.strip()})

    if response is not None:
        messages.append({"role": "assistant", "content": response.strip()})

    return tokenizer.apply_chat_template(
        messages,
        tokenize=False,
        add_generation_prompt=add_generation_prompt,
    )

def make_incomplete_chat_prompt(tokenizer, user_request, response):
    temporal = "XXXXXX"
    temp = format_chat(tokenizer, user_request, temporal)
    before_response = temp[:temp.find(temporal)]
    output = before_response + response
    return output
